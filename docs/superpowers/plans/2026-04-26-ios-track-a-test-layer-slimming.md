## Task 1: Migrate the two pure-logic avatar URL cases into the unit suite

**Files:**
- Create: `native/ios-app/Tests/Unit/FireAvatarURLTests.swift`

- [ ] **Step 1: Write the new unit-test file**

Create `native/ios-app/Tests/Unit/FireAvatarURLTests.swift` with the contents below. The two methods are lifted verbatim from `native/ios-app/Tests/Integration/FireAvatarImagePipelineTests.swift:7-33` so the assertions are byte-identical:

```swift
import XCTest
@testable import Fire

final class FireAvatarURLTests: XCTestCase {
    func testAvatarURLReplacesTemplateSizeAndResolvesRelativePath() {
        let url = fireAvatarURL(
            avatarTemplate: "/user_avatar/linux.do/alice/{size}/1_2.png",
            size: 34,
            scale: 3,
            baseURLString: "https://linux.do"
        )

        XCTAssertEqual(
            url?.absoluteString,
            "https://linux.do/user_avatar/linux.do/alice/102/1_2.png"
        )
    }

    func testAvatarURLSupportsProtocolRelativePath() {
        let url = fireAvatarURL(
            avatarTemplate: "//cdn.linux.do/user_avatar/linux.do/alice/{size}/1_2.png",
            size: 32,
            scale: 2,
            baseURLString: "https://linux.do"
        )

        XCTAssertEqual(
            url?.absoluteString,
            "https://cdn.linux.do/user_avatar/linux.do/alice/64/1_2.png"
        )
    }
}
```

Note: the imports differ from the integration original — drop `import UIKit` and `import SwiftUI` because neither symbol is referenced by the two URL cases.
